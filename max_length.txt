Q1.
There are n houses build in a line, each of which contains some value in it.

A thief is going to steal the maximal value of these houses, but he can’t steal in two adjacent houses because the owner of the stolen houses will
tell his two neighbors left and right side.

What is the maximum stolen value?

Sample Input: val[] = {6, 7, 1, 3, 8, 2, 5}

Sample Output: 20

Note: Paste the code in the answer. Also push it to git/drive
*
3 points


package training;


import java.util.Scanner;
public class max {

	    public static int maxStolenValue(int[] val) {
	        int n = val.length;
	        if (n == 0) return 0;
	        if (n == 1) return val[0];

	        int[] dp = new int[n];
	        dp[0] = val[0];
	        dp[1] = Math.max(val[0], val[1]);

	        for (int i = 2; i < n; i++) {
	            dp[i] = Math.max(val[i] + dp[i - 2], dp[i - 1]);
	        }

	        return dp[n - 1];
	    }

	    public static void main(String[] args) {
	        Scanner scanner = new Scanner(System.in);

	        System.out.print("Enter number of houses: ");
	        int n = scanner.nextInt();
	        int[] val = new int[n];

	        System.out.println("Enter the value of each house:");
	        for (int i = 0; i < n; i++) {
	            val[i] = scanner.nextInt();
	        }

	        int result = maxStolenValue(val);
	        System.out.println("Maximum stolen value: " + result);

	        scanner.close();
	    }
	}
